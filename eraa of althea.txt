        CombatMode   = Config:FindFirstChild("CombatMode")

            if s.Name == "Events" and s:FindFirstChild("LightAttack") then

            for _, s in ipairs(char:GetDescendants()) do
                if s.Name == "Events" and s:FindFirstChild("LightAttack") then
                    Events = s break
                end
            end
        end)
    end
end

-- // === POSITION SYSTEM ===
local function GetPlacementCFrame(mobPos, mobPart, mode, offset, platformY)
    local myPos, lookAt = mobPos, mobPos

    if mode == "Above" then
        local y = platformY or (mobPos.Y + offset)
        myPos = Vector3.new(mobPos.X, y, mobPos.Z)
        return CFrame.new(myPos, mobPos + Vector3.new(0.01, 0, 0))
    elseif mode == "Below" then
        myPos = Vector3.new(mobPos.X, mobPos.Y - offset, mobPos.Z)
    elseif mode == "Front" then
        local lv = (mobPart and mobPart.CFrame.LookVector) or Vector3.new(0,0,1)
        myPos = mobPos + lv * offset
    elseif mode == "Back" then
        local lv = (mobPart and mobPart.CFrame.LookVector) or Vector3.new(0,0,1)
        myPos = mobPos - lv * offset
    elseif mode == "Left" then
        local rv = (mobPart and mobPart.CFrame.RightVector) or Vector3.new(1,0,0)
        myPos = mobPos - rv * offset
    elseif mode == "Right" then
        local rv = (mobPart and mobPart.CFrame.RightVector) or Vector3.new(1,0,0)
        myPos = mobPos + rv * offset
    end

    if (myPos - lookAt).Magnitude < 0.1 then
        myPos = myPos + Vector3.new(0, 1, 0)
    end
    return CFrame.lookAt(myPos, lookAt)
end

local function FreezeVelocity()
    local hrp = GetHRP()
    if not hrp then return end
    pcall(function()
        hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
    end)
end

local function PlaceAtMob(mobPosOrPart, useTween, scope)
    local hrp = GetHRP()
    if not hrp then return end
    local mobPos, mobPart
    if typeof(mobPosOrPart) == "Vector3" then
        mobPos = mobPosOrPart
    elseif typeof(mobPosOrPart) == "Instance" and mobPosOrPart:IsA("BasePart") then
        mobPos  = mobPosOrPart.Position
        mobPart = mobPosOrPart
    else return end

    local mode, offset, platformY
    if scope == "nearest" then
        mode      = S.NearestPositionMode
        offset    = S.NearestPositionOffset
        platformY = S.NearestPlatformY
    else
        mode      = S.WolfPositionMode
        offset    = S.WolfPositionOffset
        platformY = S.WolfPlatformY
    end

    local dest = GetPlacementCFrame(mobPos, mobPart, mode, offset, platformY)
    if not dest then return end
    if useTween then
        local tw = TweenService:Create(hrp, TweenInfo.new(S.WolfFarmTweenTime, Enum.EasingStyle.Linear), {CFrame = dest})
        tw:Play(); tw.Completed:Wait()
    else
        hrp.CFrame = dest
        FreezeVelocity()
    end
end

    if not S.StickyTarget then return end

    if target.Hum and target.Hum.Health <= 0 then

    local mode, offset, platformY
    if S.StickyScope == "nearest" then

        mode      = S.WolfPositionMode
        offset    = S.WolfPositionOffset
        platformY = S.WolfPlatformY
    end

    local dest = GetPlacementCFrame(target.Part.Position, target.Part, mode, offset, platformY)
    if dest then
        hrp.CFrame = dest
        hrp.AssemblyLinearVelocity = Vector3.new(0, 0, 0)
        hrp.AssemblyAngularVelocity = Vector3.new(0, 0, 0)
    end
end)

-- // === EQUIP WEAPON ===
local function EquipWeapon()
    local char = GetCharacter()
    if not char then return end
    if char:FindFirstChildOfClass("Tool") then return end
    local client = char:FindFirstChild("Client")
    if not client then return end
    local ev  = client:FindFirstChild("Events")
    if not ev then return end
    local act = ev:FindFirstChild("ActivateWeapon")
    if not act then return end
    pcall(function()
        if act:IsA("RemoteEvent") then act:FireServer()
        elseif act:IsA("BindableEvent") then act:Fire() end
    end)
end

-- // === CHECKS ===
local function IsStunned()
    if not Config then return false end
    local stunned = false
    pcall(function()
        stunned = Config.Stunned.Value == true
            or Config.FullyStunned.Value == true
            or Config.ToggleRagdoll.Value == true
            or (Character and Character:FindFirstChild("TemporaryStun") ~= nil)
    end)
    return stunned
end

local function IsBlocking()
    if not Config then return false end
    local blocking = false
    pcall(function() blocking = Config.Blocking.Value == true end)
    return blocking
end

local function IsAttackReady()
    if not Config or not CombatMode then return false end
    local ready = false
    pcall(function()
        ready = Config.AttackCooldown.Value  == false
            and Config.CAttackCooldown.Value == false
            and Config.NoAttack.Value        == false
            and not IsStunned()
            and CombatMode.Value == true
            and Character ~= nil
            and Character:FindFirstChild("Safe Mode") == nil
            and Humanoid ~= nil
            and Humanoid.Health > 0
    end)
    return ready
end

-- // === COMBAT ===
local function ResetDebounces()
    if not Config then return end
    task.spawn(function()
        if CurrentCombo and CurrentCombo.Value > 5 then
            pcall(function() Config.CanBlock.Value = true end)
            task.wait(3)
        end
        task.wait(0.0025)
        pcall(function() Config.CAttackCooldown.Value = false end)
        pcall(function() Config.CanBlock.Value = true end)
    end)
end

local function Swing()
    if not IsAttackReady() then return false end
    if not CurrentCombo then return false end
    local anim = AttackAnimations[CurrentCombo.Value]
    if not anim then
        if Events and Events:FindFirstChild("LightAttack") then
            pcall(function() Events.LightAttack:FireServer("SecretCode22") end)
            return true
        end
        return false
    end
    pcall(function() Config.Dashing.Value = false end)
    pcall(function() Config.CAttackCooldown.Value = true end)
    if NoBlock then pcall(function() NoBlock.Value = true end) end
    anim:Play()
    anim:AdjustSpeed(1.5)
    task.delay(anim.Length / 1.5 * 0.7 - 0.4, function()
        if Events and Events:FindFirstChild("LightAttack") then
            pcall(function() Events.LightAttack:FireServer("SecretCode22") end)
        end
    end)
    local conn
    conn = anim.Stopped:Connect(function()
        if not Character then conn:Disconnect() return end
        repeat task.wait() until Character:FindFirstChild("Hitbox") == nil
        if NoBlock then pcall(function() NoBlock.Value = false end) end
        ResetDebounces()
        conn:Disconnect()
    end)
    return true
end

local function HeavySwing()
    if not IsAttackReady() then return false end
    if Events and Events:FindFirstChild("HeavyAttack") then
        pcall(function() Events.HeavyAttack:FireServer("SecretCode22") end)
        return true
    end
    return Swing()
end

local function ExecuteCombo(pattern)
    for i = 1, #pattern do
        if not S.AutoComboEnabled then break end
        local ch = pattern:sub(i, i):upper()
        if ch == "L" then
            if IsAttackReady() then Swing() end
        elseif ch == "H" then
            if IsAttackReady() then HeavySwing() end
        end
        task.wait(S.AutoComboDelay)
    end
end

local function StartBlock()
    if not Events then return end
    if IsStunned() or IsBlocking() then return end
    if not CombatMode or not CombatMode.Value then return end
    if NoBlock and NoBlock.Value then return end
    pcall(function() Events.Block:InvokeServer() end)
end

local function StopBlock()
    if not Config then return end
    pcall(function() Config.Blocking.Value = false end)
    if Events then pcall(function() Events.StopBlocking:InvokeServer() end) end
end

-- // === MOB SEARCH ===
local function FindNPCFolder()
    if S.NPCFolderName ~= "" then
        local f = Workspace:FindFirstChild(S.NPCFolderName)
        if f then return f end
    end
    for _, n in ipairs({"NPCS","NPCs","Npcs","Mobs","Enemies","Monsters","Entities"}) do
        local f = Workspace:FindFirstChild(n)
        if f then return f end
    end
    return nil
end

local function GetAliveWolves()
    local list = {}
    local keyword = S.WolfKeyword:lower()
    local folder = FindNPCFolder()
    local function Scan(parent)
        for _, obj in ipairs(parent:GetChildren()) do
            if obj:IsA("Model") and obj.Name:lower():find(keyword) then
                local hum = obj:FindFirstChildOfClass("Humanoid")
                if hum and hum.Health > 0 then
                    local part = obj.PrimaryPart or obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChildWhichIsA("BasePart")
                    if part then table.insert(list, {Model=obj, Hum=hum, Part=part}) end
                end
            elseif obj:IsA("Folder") or obj:IsA("Model") then Scan(obj) end
        end
    end
    if folder then Scan(folder) else Scan(Workspace) end
    return list
end

local function GetSortedWolves()
    local hrp = GetHRP()
    if not hrp then return {} end
    local list = GetAliveWolves()
    table.sort(list, function(a, b)
        return GetDistance(hrp.Position, a.Part.Position) < GetDistance(hrp.Position, b.Part.Position)
    end)
    return list
end

local function GetClosestWolf()
    local hrp = GetHRP()
    if not hrp then return nil, math.huge end
    local closest, best = nil, math.huge
    for _, w in ipairs(GetAliveWolves()) do
        local d = GetDistance(hrp.Position, w.Part.Position)
        if d < best then best = d; closest = w end
    end
    return closest, best
end

local function GetAllAliveMobs()
    local list = {}
    local folder = FindNPCFolder()
    local function Scan(parent)
        for _, obj in ipairs(parent:GetChildren()) do
            if obj:IsA("Model") then
                local hum = obj:FindFirstChildOfClass("Humanoid")
                if hum and hum.Health > 0 then
                    local isPlayer = false
                    for _, p in ipairs(Players:GetPlayers()) do
                        if p.Character == obj then isPlayer = true break end
                    end
                    if not isPlayer then
                        local part = obj.PrimaryPart or obj:FindFirstChild("HumanoidRootPart") or obj:FindFirstChildWhichIsA("BasePart")
                        if part then table.insert(list, {Model=obj, Hum=hum, Part=part}) end
                    end
                end
            elseif obj:IsA("Folder") then Scan(obj) end
        end
    end
    if folder then Scan(folder) else Scan(Workspace) end
    return list
end

local function GetMobNames()
    local names = {"ALL"}
    local seen = {}
    for _, mob in ipairs(GetAllAliveMobs()) do
        if not seen[mob.Model.Name] then
            seen[mob.Model.Name] = true
            table.insert(names, mob.Model.Name)
        end
    end
    return names
end

local function GetClosestFilteredMob()
    local hrp = GetHRP()
    if not hrp then return nil end
    local closest, best = nil, math.huge
    for _, mob in ipairs(GetAllAliveMobs()) do
        local nameOk = (S.NearestMobTarget == "ALL" or mob.Model.Name == S.NearestMobTarget)
        local d = GetDistance(hrp.Position, mob.Part.Position)
        if nameOk and d < best then best = d; closest = mob end
    end
    return closest
end

local function GetClosestMob()
    local hrp = GetHRP()
    if not hrp then return nil end
    local closest, best = nil, math.huge
    for _, mob in ipairs(GetAllAliveMobs()) do
        local d = GetDistance(hrp.Position, mob.Part.Position)
        if d < best then best = d; closest = mob end
    end
    return closest
end

local function GetMobsInRadius(radius, maxCount)
    local hrp = GetHRP()
    if not hrp then return {} end
    local result = {}
    local all = GetAllAliveMobs()
    table.sort(all, function(a, b)
        return GetDistance(hrp.Position, a.Part.Position) < GetDistance(hrp.Position, b.Part.Position)
    end)
    for _, mob in ipairs(all) do
        if GetDistance(hrp.Position, mob.Part.Position) <= radius then
            table.insert(result, mob)
            if maxCount and #result >= maxCount then break end
        end
    end
    return result
end

-- // === QUEST ===
local function TakeQuest()
    local hrp = GetHRP()
    if not hrp then Library:Notify('No character', 3) return end
    hrp.CFrame = CFrame.new(S.QuestPos + Vector3.new(0, S.WolfPositionOffset, 0))
    task.wait(0.6)
    hrp = GetHRP()
    if not hrp then return end
    local nearest, bestD = nil, S.ClickDistance
    for _, obj in ipairs(Workspace:GetDescendants()) do
        if obj:IsA("ClickDetector") then
            local parent = obj.Parent
            local pos
            if parent:IsA("BasePart") then pos = parent.Position
            elseif parent:IsA("Model") then pcall(function() pos = parent:GetPivot().Position end) end
            if pos then
                local allowed = not S.QuestNPCName or parent.Name == S.QuestNPCName or (parent.Parent and parent.Parent.Name == S.QuestNPCName)
                if allowed then
                    local d = GetDistance(hrp.Position, pos)
                    if d < bestD then bestD = d; nearest = obj end
                end
            end
        end
    end
    if nearest and fireclickdetector then
        fireclickdetector(nearest); Library:Notify('Quest taken!', 3)
    else
        Library:Notify('ClickDetector not found', 5)
    end
end

local ESPObjects = {}
local function ClearESP()
    for _, v in pairs(ESPObjects) do
        pcall(function() if v and v.Parent then v:Destroy() end end)
    end
    ESPObjects = {}
end

local function UpdateESP()
    ClearESP()
    if not S.ShowESP then return end
    local hrp = GetHRP()
    if not hrp then return end
    for _, w in ipairs(GetAllAliveMobs()) do
        local dist = GetDistance(hrp.Position, w.Part.Position)
        if dist <= S.ESPDistance then
            pcall(function()
                local bb = Instance.new("BillboardGui")
                bb.Size = UDim2.new(0, 140, 0, 55)
                bb.AlwaysOnTop = true
                bb.StudsOffset = Vector3.new(0, 4, 0)
                bb.Adornee = w.Part
                bb.Parent = w.Part

                local lbl = Instance.new("TextLabel")
                lbl.Size = UDim2.new(1, 0, 0.6, 0)
                lbl.BackgroundTransparency = 1
                lbl.Text = w.Model.Name .. "\nHP: " .. math.floor(w.Hum.Health) .. "/" .. math.floor(w.Hum.MaxHealth) .. " [" .. math.floor(dist) .. "m]"
                lbl.TextColor3 = Color3.fromRGB(255, 80, 80)
                lbl.TextStrokeTransparency = 0
                lbl.Font = Enum.Font.GothamBold
                lbl.TextSize = 13
                lbl.Parent = bb

                local barBg = Instance.new("Frame")
                barBg.Size = UDim2.new(0.8, 0, 0, 4)
                barBg.Position = UDim2.new(0.1, 0, 0.75, 0)
                barBg.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
                barBg.BorderSizePixel = 0
                barBg.Parent = bb

                local hpPct = math.clamp(w.Hum.Health / w.Hum.MaxHealth, 0, 1)
                local barFill = Instance.new("Frame")
                barFill.Size = UDim2.new(hpPct, 0, 1, 0)
                barFill.BackgroundColor3 = hpPct > 0.5 and Color3.fromRGB(0, 255, 0) or (hpPct > 0.25 and Color3.fromRGB(255, 255, 0) or Color3.fromRGB(255, 0, 0))
                barFill.BorderSizePixel = 0
                barFill.Parent = barBg

                table.insert(ESPObjects, bb)
            end)
        end
    end
end

-- // === EXPLOITS ===
local function VoidMob(mobData)
    if not mobData or not mobData.Model then return end
    local model = mobData.Model
    local hum   = mobData.Hum

    task.spawn(function()
        pcall(function()
            for i = 1, 30 do
                if not hum or not hum.Parent then break end
                pcall(function()
                    hum:ChangeState(Enum.HumanoidStateType.Physics)
                    hum:ChangeState(Enum.HumanoidStateType.Ragdoll)
                    hum:ChangeState(Enum.HumanoidStateType.FallingDown)
                end)
                task.wait()
            end
        end)

        pcall(function()
            for _, p in ipairs(model:GetDescendants()) do
                if p:IsA("BasePart") then
                    p.CanCollide = false
                    p.Massless = true
                    p.AssemblyLinearVelocity = Vector3.new(0, 5000, 0)
                end
            end
        end)

        task.wait(0.2)

        pcall(function()
            for _, p in ipairs(model:GetDescendants()) do
                if p:IsA("BasePart") then
                    p.CFrame = CFrame.new(p.Position.X, -10000, p.Position.Z)
                end
            end
        end)
    end)

    Library:Notify("Void: " .. model.Name, 3)
end

local ExpandedParts = {}

local function ExpandHitbox(mobData, size)
    if not mobData or not mobData.Model then return end
    local model = mobData.Model
    pcall(function()
        for _, part in ipairs(model:GetDescendants()) do
            if part:IsA("BasePart") then
                if not ExpandedParts[part] then
                    ExpandedParts[part] = {
                        Size         = part.Size,
                        Transparency = part.Transparency,
                        CanCollide   = part.CanCollide,
                        Massless     = part.Massless,
                    }
                end
                part.Size         = Vector3.new(size, size, size)
                part.Transparency = 0.6
                part.CanCollide   = false
                part.Massless     = true
            end
        end
    end)
    Library:Notify("Hitbox expanded: " .. model.Name, 3)
end

local function ResetHitboxes()
    local count = 0
    for part, orig in pairs(ExpandedParts) do
        pcall(function()
            if part and part.Parent then
                part.Size         = orig.Size
                part.Transparency = orig.Transparency
                part.CanCollide   = orig.CanCollide
                part.Massless     = orig.Massless
                count = count + 1
            end
        end)
    end
    ExpandedParts = {}
    Library:Notify("Reset " .. count .. " parts", 3)
end

local function TryInstaKill()
    local mob = GetClosestMob()
    if not mob then Library:Notify("No mobs", 3) return end

    task.spawn(function()
        EquipWeapon()
        task.wait(0.1)

        local hrp = GetHRP()
        if hrp and mob.Part and mob.Part.Parent then
            hrp.CFrame = mob.Part.CFrame * CFrame.new(0, 0, -2)
        end
        task.wait(0.1)

        for i = 1, S.InstaKillSwings do
            if not mob.Part or not mob.Part.Parent then break end
            if mob.Hum.Health <= 0 then break end
            hrp = GetHRP()
            if hrp and mob.Part and mob.Part.Parent then
                hrp.CFrame = mob.Part.CFrame * CFrame.new(0, 0, -2)
            end
            if IsAttackReady() then pcall(Swing) end
            task.wait(S.InstaKillDelay)
        end

        if mob.Hum and mob.Hum.Parent and mob.Hum.Health > 0 then
            task.wait(0.3)
            if mob.Hum.Health > 0 then
                Library:Notify("Mob alive, using Void...", 2)
                VoidMob(mob)
            end
        end

        Library:Notify("InstaKill done: " .. mob.Model.Name, 3)
    end)
end

-- // === KILL AURA ===
task.spawn(function()
    while task.wait(0.1) do
        if not S.KillAuraEnabled then continue end
        local hrp = GetHRP()
        if not hrp then continue end
        if S.AutoEquipWeapon then EquipWeapon() end

        local mobs = GetMobsInRadius(S.KillAuraRadius, 10)
        for _, mob in ipairs(mobs) do
            if not S.KillAuraEnabled then break end
            if mob.Part and mob.Part.Parent and mob.Hum.Health > 0 then
                local origCF = hrp.CFrame
                hrp.CFrame = mob.Part.CFrame * CFrame.new(0, 0, -2)
                if IsAttackReady() then pcall(Swing) end
                task.wait(S.KillAuraDelay * 0.5)
                hrp.CFrame = origCF
                task.wait(S.KillAuraDelay * 0.5)
            end
        end
    end
end)

-- // === WOLF FARM ===
local function WaitForAttackReady(timeout)
    local t = tick()
    while not IsAttackReady() do
        if tick() - t > timeout then return false end
        task.wait(0.05)
    end
    return true
end

local function WolfFarmLoop()
    while getgenv().WolfFarmRunning do
        if S.AutoEquipWeapon then EquipWeapon() end
        local wolves = GetSortedWolves()
        if #wolves == 0 then
            S.StickyTarget = nil
            if S.WolfFarmAutoQuest then
                task.spawn(function() TakeQuest(); task.wait(2) end)
            end
            task.wait(0.5)
            continue
        end
        for _, wolf in ipairs(wolves) do
            if not getgenv().WolfFarmRunning then S.StickyTarget = nil; return end
            if not wolf.Part or not wolf.Part.Parent then continue end
            if wolf.Hum.Health <= 0 then continue end

            S.StickyScope = "wolf"
            S.StickyTarget = wolf
            task.wait(0.15)

            local hit = 0
            local attempts = 0
            while hit < S.WolfFarmSwings and getgenv().WolfFarmRunning do
                attempts += 1
                if attempts > 40 then break end
                if not (wolf.Part and wolf.Part.Parent and wolf.Hum.Health > 0) then break end
                if WaitForAttackReady(1.5) then
                    pcall(Swing); hit += 1
                end
                task.wait(S.WolfFarmSwingDelay)
            end
            task.wait(0.08)
        end
        task.wait(0.05)
    end
    S.StickyTarget = nil
end

local function StartWolfFarm()
    if getgenv().WolfFarmRunning then return end
    getgenv().WolfFarmRunning = true
    S.StickyMode = true
    task.spawn(WolfFarmLoop)
    Library:Notify("Wolf Farm: STARTED", 3)
end

local function StopWolfFarm()
    getgenv().WolfFarmRunning = false
    S.StickyTarget = nil
    if not S.NearestMobEnabled then S.StickyMode = false end
    Library:Notify("Wolf Farm: STOPPED", 3)
end

-- // === NEAREST MOB FARM ===
local LastNearestAttack = 0
task.spawn(function()
    while task.wait(0.05) do
        if not S.NearestMobEnabled then
            if not getgenv().WolfFarmRunning and S.StickyScope == "nearest" then
                S.StickyTarget = nil
            end
            continue
        end
        if getgenv().WolfFarmRunning then continue end
        local hrp = GetHRP(); if not hrp then continue end
        if S.AutoEquipWeapon then EquipWeapon() end
        local mob = GetClosestFilteredMob()
        if not mob or not mob.Part or not mob.Part.Parent then
            S.StickyTarget = nil
            continue
        end

        S.StickyMode = true
        S.StickyScope = "nearest"
        S.StickyTarget = mob

        local now = tick()
        if now - LastNearestAttack >= S.AttackCooldown then
            LastNearestAttack = now
            pcall(Swing)
        end
    end
end)

        if wolf and d <= S.AttackDistance * 1.5 then
            if Config and Config.CAttackCooldown.Value == false then
                pcall(StartBlock); task.wait(0.3); pcall(StopBlock)
            end
        end
    end
end)

end)
