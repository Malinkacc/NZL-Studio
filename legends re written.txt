-- // === HELPERS ===
local function GetCharacter() return LP.Character end
local function GetHRP()
    local c = GetCharacter()
    return c and c:FindFirstChild("HumanoidRootPart")
end
local function GetHumanoid()
    local c = GetCharacter()
    return c and c:FindFirstChildOfClass("Humanoid")
end
local function GetDistance(a, b) return (a - b).Magnitude end

local function GetPartFromObject(obj)
    if not obj then return nil end
    if obj:IsA("BasePart") then return obj end
    if obj:IsA("Model") then
        return obj.PrimaryPart
            or obj:FindFirstChild("HumanoidRootPart")
            or obj:FindFirstChildWhichIsA("BasePart")
    end
    return nil
end

local function GetPos(obj)
    local p = GetPartFromObject(obj)
    if p then return p.Position end
    if obj and obj.GetPivot then
        local ok, cf = pcall(function() return obj:GetPivot() end)
        if ok and cf then return cf.Position end
    end
    return nil
end

local function FreezePlayer()
    local hrp = GetHRP()
    if not hrp then return end
    pcall(function()
        hrp.AssemblyLinearVelocity = Vector3.zero
        hrp.AssemblyAngularVelocity = Vector3.zero
    end)
end

-- // === COOLDOWN SYSTEM ===
local VisitedCooldowns = {}

local function MarkVisited(obj)
    if obj then VisitedCooldowns[obj] = tick() end
end

local function IsOnCooldown(obj, cooldownSec)
    if not obj then return false end
    local last = VisitedCooldowns[obj]
    if not last then return false end
    return (tick() - last) < cooldownSec
end

task.spawn(function()
    while task.wait(60) do
        local now = tick()
        for obj, t in pairs(VisitedCooldowns) do
            if now - t > 120 then VisitedCooldowns[obj] = nil end
        end
    end
end)

-- // === TWEEN TP ===
local CurrentTween = nil

local function CancelTween()
    if CurrentTween then
        pcall(function() CurrentTween:Cancel() end)
        CurrentTween = nil
    end
end

local function TweenTo(targetCFrame, customSpeed)
    local hrp = GetHRP()
    if not hrp then return false end
    CancelTween()
    local speed = customSpeed or S.TweenSpeed
    local dist = (hrp.Position - targetCFrame.Position).Magnitude
    if dist < 0.5 then
        FreezePlayer()
        return true
    end
    local duration = math.clamp(dist / speed, 0.05, 60)
    local tween = TweenService:Create(hrp, TweenInfo.new(duration, Enum.EasingStyle.Linear), { CFrame = targetCFrame })
    CurrentTween = tween
    tween:Play()
    local finished = false
    local conn
    conn = tween.Completed:Connect(function()
        finished = true
        if conn then conn:Disconnect() end
    end)
    local startTime = tick()
    local lastPos = hrp.Position
    local lastCheck = tick()
    local hum = GetHumanoid()
    while not finished and CurrentTween == tween do
        task.wait(0.1)
        if not hrp or not hrp.Parent then break end
        if S.AntiStuckEnabled then
            if tick() - lastCheck >= S.StuckTime then
                local moved = (hrp.Position - lastPos).Magnitude
                if moved < S.StuckDistance then
                    pcall(function()
                        if hum then hum:ChangeState(Enum.HumanoidStateType.Jumping) end
                    end)
                end
                lastPos = hrp.Position
                lastCheck = tick()
            end
        end
        if tick() - startTime > duration + 5 then CancelTween(); break end
    end
    if CurrentTween == tween then CurrentTween = nil end
    if finished then
        local hrp2 = GetHRP()
        if hrp2 then
            pcall(function()
                hrp2.CFrame = targetCFrame
                hrp2.AssemblyLinearVelocity = Vector3.zero
                hrp2.AssemblyAngularVelocity = Vector3.zero
            end)
        end
    end
    return finished
end

local function TweenToTarget(targetPos, mode, offset, customSpeed)
    mode = mode or "Above"
    offset = offset or S.TPOffset
    local destPos = targetPos
    if mode == "Above" then
        destPos = Vector3.new(targetPos.X, targetPos.Y + offset, targetPos.Z)
    end
    return TweenTo(CFrame.new(destPos, targetPos), customSpeed)
end

-- // === GENERIC ===
local function GetObjectsFromFolder(folder, useCooldown, cooldownSec)
    local list = {}
    if not folder then return list end
    for _, obj in ipairs(folder:GetChildren()) do
        local pos = GetPos(obj)
        if pos then
            local hum = obj:IsA("Model") and obj:FindFirstChildOfClass("Humanoid") or nil
            if not hum or hum.Health > 0 then
                if not useCooldown or not IsOnCooldown(obj, cooldownSec or 30) then
                    table.insert(list, {
                        Obj = obj, Pos = pos, Name = obj.Name,
                        Hum = hum, Part = GetPartFromObject(obj),
                    })
                end
            end
        end
    end
    return list
end

local function GetObjectNames(folder)
    local names = {"ALL"}
    local seen = {}
    if not folder then return names end
    for _, obj in ipairs(folder:GetChildren()) do
        if not seen[obj.Name] then
            seen[obj.Name] = true
            table.insert(names, obj.Name)
        end
    end
    return names
end

local function GetSortedObjects(folder, targetName, radius, useCooldown, cooldownSec)
    local hrp = GetHRP()
    if not hrp then return {} end
    local list = {}
    for _, o in ipairs(GetObjectsFromFolder(folder, useCooldown, cooldownSec)) do
        if targetName == "ALL" or o.Name == targetName then
            local d = GetDistance(hrp.Position, o.Pos)
            if not radius or d <= radius then table.insert(list, o) end
        end
    end
    table.sort(list, function(a, b)
        return GetDistance(hrp.Position, a.Pos) < GetDistance(hrp.Position, b.Pos)
    end)
    return list
end

local function GetSortedChests()
    return GetSortedObjects(GetChestsFolder(), S.ChestFilter, nil)

local function OpenChest(chestObj)
    if not chestObj or not chestObj.Parent then return end
    for _, d in ipairs(chestObj:GetDescendants()) do
        if d:IsA("ClickDetector") and fireclickdetector then
            pcall(function() fireclickdetector(d) end)
        elseif d:IsA("ProximityPrompt") and fireproximityprompt then
            pcall(function() fireproximityprompt(d) end)
        elseif d:IsA("BasePart") and d:FindFirstChild("TouchInterest") then
            local hrp = GetHRP()
            if hrp and firetouchinterest then
                pcall(function()
                    firetouchinterest(hrp, d, 0)
                    task.wait()
                    firetouchinterest(hrp, d, 1)
                end)
            end
        elseif d:IsA("RemoteEvent") then
            pcall(function() d:FireServer() end)
        end
    end
end

local function InstaCollectChests()
    local chests = GetSortedChests()
    if #chests == 0 then Library:Notify("No chests", 3); return end
    Library:Notify("Collecting "..#chests, 3)
    task.spawn(function()
        local hrp = GetHRP()
        local startCF = hrp and hrp.CFrame or nil
        for _, chest in ipairs(chests) do
            if not chest.Obj or not chest.Obj.Parent then continue end
            local hrpNow = GetHRP()
            if hrpNow then
                hrpNow.CFrame = CFrame.new(chest.Pos + Vector3.new(0, 2, 0))
                FreezePlayer()
                task.wait(0.05)
                OpenChest(chest.Obj)
                task.wait(0.1)
            end
        end
        if startCF and S.ChestReturnAfter then
            task.wait(0.3)
            local hrpEnd = GetHRP()
            if hrpEnd then hrpEnd.CFrame = startCF end
        end
        Library:Notify("Done!", 3)
    end)
end

local function AutoChestsLoop()
    while getgenv().LRWRunning.Chests do
        local chests = GetSortedChests()
        if #chests == 0 then
            Library:Notify("No chests left", 3)
            getgenv().LRWRunning.Chests = false
            break
        end
        for _, chest in ipairs(chests) do
            if not getgenv().LRWRunning.Chests then break end
            if chest.Obj and chest.Obj.Parent then
                TweenToTarget(chest.Pos, "Above", S.TPOffset)
                task.wait(0.15)
                OpenChest(chest.Obj)
                task.wait(S.ChestDelay)
            end
        end
        task.wait(0.3)
    end
end

-- // === MOBS HOVER ===
local function GetMobPlacement(mob, mode, offset, platformY)
    mode = mode or S.MobPositionMode
    offset = offset or S.MobPositionOffset
    local mobPos = mob.Pos
    if mode == "Above" then
        local y = platformY or (mobPos.Y + offset)
        return CFrame.new(Vector3.new(mobPos.X, y, mobPos.Z), mobPos + Vector3.new(0.01, 0, 0))
    elseif mode == "Back" and mob.Part then
        local lv = mob.Part.CFrame.LookVector
        return CFrame.lookAt(mobPos - lv * offset, mobPos)
    elseif mode == "Front" and mob.Part then
        local lv = mob.Part.CFrame.LookVector
        return CFrame.lookAt(mobPos + lv * offset, mobPos)
    elseif mode == "Side" and mob.Part then
        local rv = mob.Part.CFrame.RightVector
        return CFrame.lookAt(mobPos + rv * offset, mobPos)
    end
    return CFrame.new(mobPos + Vector3.new(0, offset, 0), mobPos)
end

local HoverActive = false
local HoverTarget = nil
local HoverParams = { mode = "Above", offset = 6, platformY = nil }

RunService.Heartbeat:Connect(function()
    if not HoverActive or not HoverTarget then return end
    pcall(function()
        local mob = HoverTarget
        if not mob.Obj or not mob.Obj.Parent then return end
        local hrp = GetHRP()
        if not hrp then return end
        local newPos = GetPos(mob.Obj)
        if newPos then mob.Pos = newPos end
        hrp.CFrame = GetMobPlacement(mob, HoverParams.mode, HoverParams.offset, HoverParams.platformY)
        hrp.AssemblyLinearVelocity = Vector3.zero
        hrp.AssemblyAngularVelocity = Vector3.zero
    end)
end)

local function AutoMobsLoop()
    while getgenv().LRWRunning.Mobs do
        local mobs = GetSortedObjects(GetMobsFolder(), S.MobsTarget, S.MobsRadius)
        if #mobs == 0 then task.wait(0.5); continue end
        for _, mob in ipairs(mobs) do
            if not getgenv().LRWRunning.Mobs then break end
            if mob.Obj and mob.Obj.Parent and (not mob.Hum or mob.Hum.Health > 0) and not IsBoss(mob.Name) then
                TweenTo(GetMobPlacement(mob, S.MobPositionMode, S.MobPositionOffset, S.MobPlatformY))
                CancelTween()
                HoverParams = { mode = S.MobPositionMode, offset = S.MobPositionOffset, platformY = S.MobPlatformY }
                HoverTarget = mob
                HoverActive = true
                local t = tick()
                local minTime = math.max(S.MobMinTimeOnTarget, S.MobHoverTime)
                while tick() - t < minTime and getgenv().LRWRunning.Mobs do
                    if not mob.Obj or not mob.Obj.Parent then break end
                    if mob.Hum and mob.Hum.Health <= 0 then
                        if tick() - t >= S.MobMinTimeOnTarget then break end
                    end
                    task.wait(0.1)
                end
                HoverActive = false
                HoverTarget = nil
                task.wait(0.1)
            end
        end
        task.wait(0.05)
    end
    HoverActive = false
    HoverTarget = nil
end

    getgenv().LRWRunning.Mobs = true
    task.spawn(AutoMobsLoop)
end

-- // === BOSSES ===
local function GetAllBosses()
    local list = {}
    local folder = GetMobsFolder()
    if not folder then return list end
    for _, obj in ipairs(folder:GetDescendants()) do
        if obj:IsA("Model") and IsBoss(obj.Name) then
            local hum = obj:FindFirstChildOfClass("Humanoid")
            local part = GetPartFromObject(obj)
            if part then
                table.insert(list, {
                    Obj = obj, Pos = part.Position, Name = obj.Name,
                    Hum = hum, Part = part,
                    Alive = (not hum) or hum.Health > 0,
                })
            end
        end
    end
    return list
end

local function GetAliveBossNames()
    local list = {}
    local seen = {}
    for _, b in ipairs(GetAllBosses()) do
        if b.Alive and not seen[b.Name] then
            seen[b.Name] = true
            table.insert(list, b.Name)
        end
    end
    return list
end

local function GetSortedBosses()
    local hrp = GetHRP()
    if not hrp then return {} end
    local list = {}
    for _, b in ipairs(GetAllBosses()) do
        if b.Alive and (S.BossName == "ALL" or b.Name == S.BossName) then
            table.insert(list, b)
        end
    end
    local priorityMap = {}
    for i, name in ipairs(S.BossPriorities) do priorityMap[name] = i end
    table.sort(list, function(a, b)
        local pa = priorityMap[a.Name] or 999
        local pb = priorityMap[b.Name] or 999
        if pa ~= pb then return pa < pb end
        return GetDistance(hrp.Position, a.Pos) < GetDistance(hrp.Position, b.Pos)
    end)
    return list
end

local function AutoBossLoop()
    while getgenv().LRWRunning.Boss do
        local bosses = GetSortedBosses()
        if #bosses == 0 then task.wait(2); continue end
        for _, boss in ipairs(bosses) do
            if not getgenv().LRWRunning.Boss then break end
            if boss.Obj and boss.Obj.Parent and (not boss.Hum or boss.Hum.Health > 0) then
                TweenTo(GetMobPlacement(boss, S.BossPositionMode, S.BossPositionOffset, nil))
                CancelTween()
                HoverParams = { mode = S.BossPositionMode, offset = S.BossPositionOffset, platformY = nil }
                HoverTarget = boss
                HoverActive = true

                local t = tick()
                local hitsDone = 0
                local maxTime = math.max(S.BossHoverTime, S.BossMinHits * (S.AutoAttackDelay or 0.4))

                while getgenv().LRWRunning.Boss do
                    if not boss.Obj or not boss.Obj.Parent then break end
                    if boss.Hum and boss.Hum.Health <= 0 then
                        SessionStats.BossesKilled = SessionStats.BossesKilled + 1
                        break
                    end
                    local elapsed = tick() - t
                    if elapsed >= maxTime then break end
                    if elapsed >= S.BossHoverTime and hitsDone >= S.BossMinHits then break end
                    task.wait(0.1)
                    hitsDone = hitsDone + 1
                end

                HoverActive = false
                HoverTarget = nil
                task.wait(0.3)
            end
        end
        task.wait(0.5)
    end
    HoverActive = false
    HoverTarget = nil

local function FindPickaxe()
    local char = GetCharacter()
    if not char then return nil end
    local equipped = char:FindFirstChildOfClass("Tool")
    if equipped then
        if S.PickaxeChoice ~= "AUTO" and equipped.Name == S.PickaxeChoice then return equipped end
        if S.PickaxeChoice == "AUTO" and IsPickaxe(equipped.Name) then return equipped end
    end
    local bp = LP:FindFirstChild("Backpack")
    if not bp then return nil end
    if S.PickaxeChoice ~= "AUTO" then
        local t = bp:FindFirstChild(S.PickaxeChoice)
        if t then return t end
    end
    local priority = {
        "DragonStonePickaxe", "StarForgedPickaxe", "ManaForgedPickaxe",
        "ManaCrystalPickaxe", "CrystalPickaxe", "RunePickaxe",
        "FrenzyPickaxe", "AdamantPickaxe", "BlackPickaxe",
        "SteelPickaxe", "IronPickaxe", "BronzePickaxe",
    }
    for _, name in ipairs(priority) do
        local t = bp:FindFirstChild(name)
        if t then return t end
    end
    for _, tool in ipairs(bp:GetChildren()) do
        if tool:IsA("Tool") and IsPickaxe(tool.Name) then return tool end
    end
    return nil
end

local function EquipPickaxe()
    local char = GetCharacter()
    if not char then return false end
    local equipped = char:FindFirstChildOfClass("Tool")
    if equipped then
        if S.PickaxeChoice == "AUTO" and IsPickaxe(equipped.Name) then return true end
        if S.PickaxeChoice ~= "AUTO" and equipped.Name == S.PickaxeChoice then return true end
        equipped.Parent = LP:FindFirstChild("Backpack")
        task.wait(0.1)
    end
    local pick = FindPickaxe()
    if not pick then return false end
    if pick.Parent ~= char then pick.Parent = char end
    return true
end

local function AutoOresLoop()
    while getgenv().LRWRunning.Ores do
        if S.AutoEquipPickaxe then EquipPickaxe() end
        local ores = GetSortedObjects(GetOresFolder(), S.OresTarget, S.OresRadius, true, S.OresCooldown)
        if #ores == 0 then task.wait(2); continue end
        for _, ore in ipairs(ores) do
            if not getgenv().LRWRunning.Ores then break end
            if ore.Obj and ore.Obj.Parent then
                TweenTo(GetSimplePlacement(ore.Pos, S.OresPositionOffset))
                task.wait(0.15)
                if S.AutoEquipPickaxe then EquipPickaxe(); task.wait(0.1) end
                local startTime = tick()
                local hits = 0
                while hits < S.OresHitsCount and (tick() - startTime) < S.OresMinTime + 10 do
                    if not getgenv().LRWRunning.Ores then break end
                    if not ore.Obj or not ore.Obj.Parent then break end
                    local hrp = GetHRP()
                    if hrp then
                        hrp.CFrame = GetSimplePlacement(ore.Pos, S.OresPositionOffset)
                        FreezePlayer()
                    end
                    MineOre(ore.Obj)
                    hits = hits + 1
                    task.wait(S.OresHitDelay)
                end
                while tick() - startTime < S.OresMinTime do
                    if not getgenv().LRWRunning.Ores then break end
                    task.wait(0.1)
                end
                MarkVisited(ore.Obj)
                task.wait(0.1)
            end
        end
        task.wait(0.3)
    end
end

-- // === SERVER HOP ===
local function HttpRequest(url)
    local req = (syn and syn.request) or (http and http.request) or http_request or request
    if req then
        local ok, response = pcall(function() return req({ Url = url, Method = "GET" }) end)
        if ok and response then return response.Body or response.body end
    end
    local ok, body = pcall(function() return game:HttpGet(url) end)
    if ok then return body end
    return nil
end

local function FetchServers(cursor)
    local placeId = game.PlaceId
    local url = "https://games.roblox.com/v1/games/"..placeId.."/servers/Public?sortOrder=Asc&limit=100"
    if cursor then url = url.."&cursor="..cursor end
    local body = HttpRequest(url)
    if not body then return nil end
    local ok, data = pcall(function() return HTTP:JSONDecode(body) end)
    if not ok then return nil end
    return data
end

local function GetGoodServers()
    local good = {}
    local cursor = nil
    local pageCount = 0
    repeat
        local data = FetchServers(cursor)
        if not data or not data.data then break end
        for _, srv in ipairs(data.data) do
            local players = srv.playing or 0
            local maxP = srv.maxPlayers or 0
            if players >= S.HopMinPlayers and players <= S.HopMaxPlayers and players < maxP then
                if srv.id ~= game.JobId then
                    local visited = false
                    if S.HopAvoidVisited then
                        for _, vid in ipairs(getgenv().LRWVisitedServers) do
                            if vid == srv.id then visited = true; break end
                        end
                    end
                    if not visited then table.insert(good, srv) end
                end
            end
        end
        cursor = data.nextPageCursor
        pageCount = pageCount + 1
    until not cursor or pageCount >= 5
    return good
end

local function ServerHop()
    Library:Notify("Searching servers...", 3)
    local servers = GetGoodServers()
    if #servers == 0 then
        if S.HopAvoidVisited then
            getgenv().LRWVisitedServers = {game.JobId}
            servers = GetGoodServers()
        end
        if #servers == 0 then Library:Notify("No servers found", 4); return end
    end
    local picked = servers[math.random(1, #servers)]
    table.insert(getgenv().LRWVisitedServers, picked.id)
    Library:Notify("Hopping ("..picked.playing.."/"..picked.maxPlayers..")", 3)
    if queue_on_teleport and S.HopScriptURL ~= "" then
        pcall(function() queue_on_teleport("loadstring(game:HttpGet('"..S.HopScriptURL.."'))()") end)
    end
    pcall(function() TS:TeleportToPlaceInstance(game.PlaceId, picked.id, LP) end)
end

    while task.wait(5) do
        if not S.HopAutoEnabled then continue end
        local shouldHop = false
        if S.HopAutoTrigger == "Timer" then

        elseif S.HopAutoTrigger == "NoChests" then

        elseif S.HopAutoTrigger == "NoOres" then
            local f = GetOresFolder()
            if f and #f:GetChildren() == 0 then shouldHop = true end
        elseif S.HopAutoTrigger == "NoPlants" then
            local f = GetPlantsFolder()
            if f and #f:GetChildren() == 0 then shouldHop = true end
        elseif S.HopAutoTrigger == "NoBosses" then
            if #GetAliveBossNames() == 0 then shouldHop = true end
        elseif S.HopAutoTrigger == "PlayersFull" then
            if #Players:GetPlayers() >= S.HopMaxPlayers then shouldHop = true end
        end
        if shouldHop then ServerHop(); break end
    end
end)

    StatusLabel.Position = UDim2.new(0, 5, 0, 22)

    StatusLabel.TextColor3 = Color3.fromRGB(220, 220, 240)
    StatusLabel.Font = Enum.Font.RobotoMono
    StatusLabel.TextSize = 11
    StatusLabel.TextXAlignment = Enum.TextXAlignment.Left
    StatusLabel.TextYAlignment = Enum.TextYAlignment.Top

    while task.wait(1) do
        StatusFrame.Visible = S.ShowStatusGUI
        if S.ShowStatusGUI then
            pcall(function()
                local elapsed = tick() - SessionStats.StartTime
                local mins = math.floor(elapsed / 60)
                local secs = math.floor(elapsed % 60)
                local status = {}
                if getgenv().LRWRunning.Chests then table.insert(status, "Chests") end
                if getgenv().LRWRunning.Mobs   then table.insert(status, "Mobs") end

                if S.AutoAbility   then table.insert(status, "Abi") end
                if S.FlyEnabled    then table.insert(status, "Fly") end
                if S.HopAutoEnabled then table.insert(status, "AHop") end
                local statusText = #status == 0 and "Idle" or table.concat(status, ", ")

                local aliveBosses = GetAliveBossNames()

                    end

                    "Time: %02d:%02d\nGold: %d  Items: %d\nBosses killed: %d\nCursed: %d\nStatus: %s\n\n[Alive bosses: %d]\n%s",

                    SessionStats.GoldGained, SessionStats.ItemsCollected,
                    SessionStats.BossesKilled, SessionStats.CursedFingersGrabbed,
                    statusText, #aliveBosses, bossText

end)
